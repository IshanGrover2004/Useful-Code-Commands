name: CI/CD

on:
  push:
    branches:
      - main

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-gcc@v2
      - run: g++ -o modmanager modmanager.cpp

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-gcc@v2
      - run: g++ -o modmanager modmanager.cpp
      - run: ./modmanager
      - run: ctest -V

  deploy:
    runs-on: ubuntu-latest
    needs:
      - build
      - test
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-git@v2
      - run: git add .
      - run: git commit -m "Deploying new release"
      - run: git push origin main
